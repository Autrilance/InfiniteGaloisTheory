	\lean{AlgHom.toAlgEquiv_fieldRange}
	\leanok
	Let $ K, L, E $ be fields and $ L $ and $ E $ are $ K $-algebras, $\sigma : L \to E $ is an $ K $-algebra homomorphism, them there is a $ K $-algebra isomomorphism between $ L $ and $ E $.
\end{lemma}
